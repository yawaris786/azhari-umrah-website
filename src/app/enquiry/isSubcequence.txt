

write a function that takes 2 strings and checks whether the characters in the first string form a subcequence of the characters in the 2nd string.

 

Eg - isSubcequence('hello' , 'hello world') // true

isSubcequence('sing' , 'sting') // true

isSubcequence('abc' , 'acb') // false


function isSubsequence(str1, str2) {
  let i = 0; // pointer for str1
  let j = 0; // pointer for str2

  // Loop until we reach the end of str2
  while (j < str2.length) {
    // Check if characters at i and j match
    if (str1[i] === str2[j]) {
      i++; // move pointer for str1
    }
    // If we have found all characters in str1, return true
    if (i === str1.length) {
      return true;
    }
    j++; // move pointer for str2
  }

  // If we complete the loop without finding all characters, return false
  return false;
}

// Test cases
console.log(isSubsequence('hello', 'hello world')); // true
console.log(isSubsequence('sing', 'sting')); // true
console.log(isSubsequence('abc', 'acb')); // false
